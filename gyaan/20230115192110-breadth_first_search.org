  vector<int> bfsOfGraph(int V, vector<int> adj[]) {
    // Code here
    vector<bool> vis(V, false);
    vector<int> ans;
        
    queue<int> q;
    vis[0] = true;
    q.push(0);
        
    while(!q.empty()) {
      int f = q.front();
      ans.push_back(f);
      q.pop();
            
      for(int i : adj[f]) {
        if(vis[i]) continue;
        vis[i] = true;
        q.push(i);
      }
    }
    return ans;
  }
#+end_src
